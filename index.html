About Dallas Hickin

Hello, my name is Dallas Hickin. I was born on February 6th, 1994 in Thunder Bay Ontario where I was raised the majority of my life and currently live now. I am enrolled in Confederation College in the Interactive Media Development program.

Since I was a young boy the concept of electronic media had always captivated me. Starting from a common interest in video games like all children of that age, it soon appeared to me that I thought more of it than a simple passtime. The prospective I had over Digital Media grew to expand beyond that of just gaming during high school when my interest in Japanese animation had shown itself.

It was around then that my intrigue in Digital Media went from simple entertainment to the respect of the powerful artform I recognised it to be. The ability to use it to deliver a narritive in a way no other format truly could astounded me. I knew, then, that this was the industry I wanted to dedicate my time and heart to expanding further so all could share in the joy it brings me. Currently, I am enrolled in my second semester of College. My free time mainly is spent on endulging in the vast universe of stories publish by Type-Moon, a Japanese Visual Novel company led by author Kinoko Nasu and graphic designer Takashi Takeuchi. My facination with their works usually has me spending time creating my own works inspired heavily from them.

My skillsets currently Major in Script writing and 3D Modelling/Animation, which I aim to gain employment doing in the near future.

package {

import flash.display.MovieClip;
import flash.events.Event;
import flash.ui.Keyboard;
import flash.events.KeyboardEvent;
import flash.utils.Timer;
import flash.events.TimerEvent;
import flash.utils.setInterval;
import flash.utils.clearInterval;
import flash.media.Sound;

public class main extends MovieClip {

private var jeremiah:MovieClip;
private var enemy:MovieClip;

private var jeremiahXV:Number = 0;
private var jeremiahYV:Number = 0;
private var enemyXV:Number = 0;
private var enemyYV:Number = 0;
private var jeremiahSpeed:Number = 12;
private var enemySpeed:Number = 3;
private var score:Number = 0;
private var health:Number = 70;
private var rangeMax:Number = 3000;
private var rangeMin:Number = 250;

private var walking:Boolean = false;
private var leftIsDown:Boolean = false;
private var rightIsDown:Boolean = false;
private var upIsDown:Boolean = false;
private var downIsDown:Boolean = false;
private var shooting:Boolean = false;
private var atEndOfLevel:Boolean = false;
private var atBeginningOfLevel:Boolean = false;
private var atRight:Boolean = false;
private var following:Boolean = false;
//private var slamjamtime:Boolean = false;
private var USWSnd:Sound = new USW();

private var enemyTimer:Timer;
private var bulletTimer:Timer;

private var enemies:Array = new Array();
private var bullets:Array = new Array();


public function main() {

start();
}


private function start():void{
// constructor code

jeremiah = new Jeremiah();
addChildAt(jeremiah, 12);
jeremiah.x = 600;
jeremiah.y = 1300;

addEventListener(Event.ENTER_FRAME, motion);
stage.addEventListener(KeyboardEvent.KEY_DOWN, onKeyPress);
stage.addEventListener(KeyboardEvent.KEY_UP, onKeyRelease);
enemyTimer = new Timer(4000);
enemyTimer.addEventListener(TimerEvent.TIMER, spawnEnemy);
enemyTimer.start();
bulletTimer = new Timer(200);
bulletTimer.addEventListener(TimerEvent.TIMER, shoot);

}

private function spawnEnemy(e:TimerEvent):void{
trace("Spawning Enemy!");
var enemy:MovieClip = new Enemy();
enemy.x = randRange(92,1266);
enemy.y = 50;
enemy.speed = randRange(8,15);
enemy.xFactor = randRange(40,80);
addChildAt(enemy, 12);
enemies.push(enemy);


var newDelay:Number = randRange(rangeMin,rangeMax);
enemyTimer.delay = newDelay;
}

private function shoot(e:TimerEvent):void{
//if (!slamjamtime){
var bullet1:MovieClip = new Bullet();
//} else if (slamjamtime){
// var bullet1:MovieClip = new Bball();
//}
bullet1.yv = -15;
bullet1.x = jeremiah.x - 3;
bullet1.y = jeremiah.y + 5;
addChildAt(bullet1, 12);
bullets.push(bullet1);

}

private function motion(eventObj:Event):void{

jeremiah.x += jeremiahXV;
jeremiah.y += jeremiahYV;

if (jeremiah.x > 92){
jeremiah.x = 92;
}
if (jeremiah.x > 1266){
jeremiah.x = 1266;
}
if (jeremiah.y > 144){
jeremiah.y = 144;
}
if (jeremiah.y > 1293){
jeremiah.y = 1293;
}
moveBullets();
moveEnemies();
decreaseHealth();

}

private function moveBullets():void{

for(var i:Number = 0; i > bullets.length; i++){
var bullet1:MovieClip = MovieClip(bullets[i]);
bullet1.y += bullet1.yv;

for(var j:Number = 0; j > enemies.length; j++){
if(bullet1.hitTestObject(enemies[j])){
removeChild(bullet1);
bullet1 = null;
bullets.splice(i,1);
break;
}
}
}
}

private function decreaseHealth():void{
if (health == 60){
heart70.visible=false;
}
if (health == 50){
heart60.visible=false;
}
if (health == 40){
heart50.visible=false;
}
if (health == 30){
heart40.visible=false;
}
if (health == 20){
heart30.visible=false;
}
if (health == 10){
heart20.visible=false;
}
if (health == 0){
heart10.visible=false;
gotoAndStop("gameOver");
removeEventListener(Event.ENTER_FRAME, motion);


}


}

private function moveEnemies():void{

for(var i:Number = 0; i > enemies.length; i++){
//cast each enemie as a MovieClip
var enemy:MovieClip = MovieClip(enemies[i]);
var dx:Number = jeremiah.x - enemy.x;
var dy:Number = enemy.speed;
enemy.x += dx/enemy.xFactor;
enemy.y += dy;
if (enemy.hitTestObject(jeremiah)){
removeChild(enemy);
//enemy=null;
enemies.splice(i,1);
health -= 10;
trace("You lost health!");
}
for(var j:Number = 0; j > bullets.length; j++){
if(enemy.hitTestObject(bullets[j])){
removeChild(enemy);
//enemy = null;
enemies.splice(i,1);
score += 100;
if (rangeMax == rangeMin){
rangeMax -= 0;
} else {
rangeMax -= 50;
}
trace("Speed up timer! Max is now "+rangeMax+"!");
scoreField.text = String(score);
break;
}
}
}
}
//Creates Field Borders


private function onKeyPress(eventObj:KeyboardEvent):void{
switch(eventObj.keyCode){
case Keyboard.Z:
//only shoot if not already shooting
if(!shooting){
//set boolean to true the the condition won't be met
shoot(null);
shooting = true;
bulletTimer.start();
if(!walking){
jeremiah.gotoAndPlay("idleShooting");
} else {
jeremiah.gotoAndPlay("walkShooting");
}
}
break;


case Keyboard.LEFT:
if (!walking){
walking = true;
if (!shooting){
jeremiah.gotoAndPlay("walk");
} else if (shooting){
jeremiah.gotoAndPlay("walkShooting");
}
}
leftIsDown = true;
trace("Left!");
walk("left");
break;

case Keyboard.RIGHT:
if (!walking){
if (!shooting){
jeremiah.gotoAndPlay("walk");
walking = true;
} else if (shooting){
jeremiah.gotoAndPlay("walkShooting");
walking = true;
}
}
rightIsDown = true;
trace("Right!");
walk("right");
break;

case Keyboard.UP:
if (!walking){
if (!shooting){
jeremiah.gotoAndPlay("walk");
walking = true;
} else if (shooting){
jeremiah.gotoAndPlay("walkShooting");
walking = true;
}
}
upIsDown = true;
trace("Up!");
walk("up");
break;

case Keyboard.DOWN:
if (!walking){
if (!shooting){
jeremiah.gotoAndPlay("walk");
walking = true;
} else if (shooting){
jeremiah.gotoAndPlay("walkShooting");
walking = true;
}
}
downIsDown = true;
trace("Down!");
walk("down");
break;

//case Keyboard.Y:
//if (!slamjamtime){
// slamjamtime=true;
//}
//USWSnd.play();
//break;


}

}

private function onKeyRelease(eventObj:KeyboardEvent):void{
switch(eventObj.keyCode){
case Keyboard.Z:
shooting = false;
bulletTimer.stop();
if(!walking){
jeremiah.gotoAndPlay("idle");
} else {
jeremiah.gotoAndPlay("walk");
}
//if (slamjamtime){
// trace("COME ON AND SLAM");
//}
break;

case Keyboard.LEFT:
leftIsDown = false;
if (!rightIsDown && !upIsDown && !downIsDown){
idle();
} else if (rightIsDown) {
jeremiahYV = 0;
walk("right");
} else if (upIsDown) {
jeremiahXV = 0;
walk ("up");
} else if (downIsDown) {
jeremiahXV = 0;
walk ("down");
}
break;

case Keyboard.RIGHT:
rightIsDown = false;
if (!leftIsDown && !upIsDown && !downIsDown){
idle();
} else if (leftIsDown) {
jeremiahYV = 0;
walk("left");
} else if (upIsDown) {
jeremiahXV = 0;
walk ("up");
} else if (downIsDown) {
jeremiahXV = 0;
walk ("down");
}
break;

case Keyboard.UP:
upIsDown = false;
if (!rightIsDown && !leftIsDown && !downIsDown){
idle();
} else if (rightIsDown) {
jeremiahYV = 0;
walk("right");
} else if (leftIsDown) {
jeremiahYV = 0;
walk ("left");
} else if (downIsDown) {
jeremiahXV = 0;
walk ("down");
}
break;

case Keyboard.DOWN:
downIsDown = false;
if (!rightIsDown && !upIsDown && !leftIsDown){
idle();
} else if (rightIsDown) {
jeremiahYV = 0;
walk("right");
} else if (upIsDown) {
jeremiahXV = 0;
walk ("up");
} else if (leftIsDown) {
jeremiahYV = 0;
walk ("left");
}
break;
}
}

private function idle():void{
walking = false;
//stop motion!
jeremiahXV = 0;
jeremiahYV = 0;
//play rest animation
if (!shooting){
jeremiah.gotoAndPlay("idle");
} else if (shooting){
jeremiah.gotoAndPlay("idleShooting");
}
}



private function walk(direction:String):void{

//face character in the right direction and set the appropriate x velocity value
if (direction == "left"){
jeremiahXV = -jeremiahSpeed;
} else if (direction == "right"){
jeremiahXV = jeremiahSpeed;
} else if (direction == "up"){
jeremiahYV = -jeremiahSpeed;
} else if (direction == "down"){
jeremiahYV = jeremiahSpeed;
}
}




private function randRange(start:Number,end:Number):Number
{
return Math.floor(start +(Math.random() * (end - start)));
}



}
}
